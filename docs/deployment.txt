# Setup
git clone https://github.com/yourusername/letta-acp-bridge
cd letta-acp-bridge
python -m venv venv
source venv/bin/activate  # or venv\Scripts\activate on Windows
pip install -r requirements.txt

# Configure
cp .env.example .env
# Edit .env with your Letta server URL

# Run locally
python -m src.acp_server

# Test in another terminal
python examples/test_client.py
